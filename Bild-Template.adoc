// Zweites + drittes Feld oder width/height = Bildgröße

.Eine klickbare Grafik
image::diagramm.png[Diagramm, width=500, link=https://example.org]

// ----- Mit Rolle für CSS- oder PDF-Themeneinstellungen --
image::banner.png[Banner, width=600, role=banner-image]
